\newcommand{\tweetmedia}[1]{
    \begin{tikzpicture}
        \def\ig{%
        \includegraphics[width=\textwidth,keepaspectratio]{#1}}
    \node [inner sep=0pt](mypicture) at (0,0) {\phantom{\ig}};
    \clip[rounded corners=2mm] ($(mypicture.south west)+(\bord,\bord)$) rectangle ($(mypicture.north east)-(\bord,\bord)$);
    \node[inner sep=0pt](mypicture) at (0,0) {\ig};
    \end{tikzpicture}
}

\newcommand{\threadline}[1]{

        \begin{tikzpicture}
            \def\ig{%
            \textcolor{black!40}{\rule{3mm}{#1cm}}}
          \node [inner sep=0pt](mypicture) at (0,0) {\phantom{\ig}};
          \clip[rounded corners=0.8mm] ($(mypicture.south west)+(\bord,\bord)$) rectangle ($(mypicture.north east)-(\bord,\bord)$);
          \node[inner sep=0pt](mypicture) at (0,0) {\ig};
       \end{tikzpicture}
    \end{center}
}
